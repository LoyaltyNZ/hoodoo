def initialize(amqp_uri, name, options = {})
  super amqp_uri, options
  @request_endpoint = "#{name}"
  @response_endpoint = "#{name}.#{@endpoint_id}"
  @queue_options = options[:queue_options] || {:exclusive => false, :auto_delete => false}
end
